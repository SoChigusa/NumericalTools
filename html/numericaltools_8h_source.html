212 double NTools::SimpsonIntegrator::integrate(std::function<double(double)> arg_f,
213  double arg_x0, double arg_x1, double arg_dx)
214 {
215  double res = 0.;
216  double xmd;
217  double xlg = arg_x0;
218  double fsm, fmd;
219  double flg = arg_f(arg_x0);
220  while(xlg < arg_x1) {
221  xmd = xlg + 0.5 * arg_dx;
222  xlg = xlg + arg_dx;
223  fsm = flg;
224  fmd = arg_f(xmd);
225  flg = arg_f(xlg);
226  res += (fsm+4.*fmd+flg)*arg_dx/6.;
227  }
228  return res;

244 {
245  const int nmesh = arg_x.size();
246  if(arg_f.size() != nmesh) {
247  throw "Mismatch between # of mesh of x and y\n";
248  }
249 
250  double res = 0.;
251  if(arg_flag == "Simpson") {
252  if(nmesh % 2 == 0) {
253  throw "Odd # of date including start and end point is needed\n";
254  } else if(nmesh < 3) {
255  throw "Too small # of mesh\n";

257

274  }

290 double NTools::BooleIntegrator::integrate(double arg_dx, std::vector<double> arg_f) {
291  double res = 0.;
292  const int nmesh = arg_f.size();
293  static const double f29o24 = 29./24.;
294  static const double f71o24 = 71./24.;
295  static const double f35o24 = 35./24.;
296  static const double to45 = 2./45.;
297  res += f29o24*arg_f[0]+f71o24*arg_f[1]+f35o24*arg_f[2]+0.375*arg_f[3]; // 1st, 2nd, 3rd order
298  res += 0.375*arg_f[nmesh-4]+f35o24*arg_f[nmesh-3]+f71o24*arg_f[nmesh-2]+f29o24*arg_f[nmesh-1];
299  for(int i = 2; i < nmesh-2; ++i) {// 1st order
300  res += to45*(7.*arg_f[i-2] + 32.*arg_f[i-1] + 12.*arg_f[i] + 32.*arg_f[i+1] + 7.*arg_f[i+2]);
301  }
302  res *= 0.25 * arg_dx; // treat over-counting & spacing
303  return res;

314 {
315  const int n = arg_x.size()-1;
316  if(arg_y.size() != n+1) {
317  throw "Mismatch between # of x and y points\n";
318  }

320 #ifdef _NTOOLS_CHECK_ASCEND
321  else {
322  std::vector<double> check;
323  std::copy(arg_x.begin(), arg_x.end(), std::back_inserter(check));
324  std::sort(check.begin(), check.end());
325  if(!std::equal(arg_x.begin(), arg_x.end(), check.begin())) {
326  throw "Data should be ascending order in x\n";
327  }
328  }
329 #endif
330 
331  endpoint.resize(arg_x.size());
332  std::copy(arg_x.begin(), arg_x.end(), endpoint.begin());
333  std::vector<double> h(n), b(n), d(n), g(n), u(n);
334  p.resize(n);
335  q.resize(n);
336  r.resize(n+1);
337  s.resize(n);
338 
339  int i;
340  for(i=0; i<n; i++) { h[i]=arg_x[i+1]-arg_x[i]; }
341  for(i=1; i<n; i++) {
342  b[i]=2.*(h[i]+h[i-1]);
343  d[i]=3.*((arg_y[i+1]-arg_y[i])/h[i] - (arg_y[i]-arg_y[i-1])/h[i-1]);
344  }
345  g[1]=h[1]/b[1];
346  for(i=2; i<n-1; i++) { g[i]=h[i]/(b[i]-h[i-1]*g[i-1]); }
347  u[1]=d[1]/b[1];
348  for(i=2; i<n; i++) { u[i]=(d[i]-h[i-1]*u[i-1])/(b[i]-h[i-1]*g[i-1]); }
349  r[0]=0; r[n]=0;
350  r[n-1]=u[n-1];
351  for(i=n-2; i>=1; i--) { r[i]=u[i]-g[i]*r[i+1]; }
352  for(i=0; i<n; i++) {
353  p[i]=arg_y[i];
354  q[i]=(arg_y[i+1]-arg_y[i])/h[i]-h[i]*(r[i+1]+2.*r[i])/3.;
355  s[i]=(r[i+1]-r[i])/(3.*h[i]);
356  }

398 double NTools::SplineInterpolator::integrate(double arg_x0, double arg_x1)
399 {
400  if(arg_x0 < endpoint.front() || arg_x0 > endpoint.back() ||
401  arg_x1 < endpoint.front() || arg_x1 > endpoint.back()) {
402  throw "Point is not between the interpolation region in SplineInterpolator::integrate\n";
403  } else if(arg_x0 > arg_x1) {
404  throw "Integration with x0 > x1 is not supported\n";
405  }
406 
407  bool fsum = 0;
408  double res = 0.;
409  double sub_x0, sub_x1;
410  const int n = endpoint.size()-1;
411  for(int i = 0; i < n; i++) {
412  if(fsum == 0) {
413  if(endpoint[i+1] > arg_x0) { fsum = 1; }
414  }
415  if(fsum == 1) {
416  sub_x0 = std::max(endpoint[i], arg_x0);
417  sub_x1 = std::min(endpoint[i+1], arg_x1);
418  auto F = [=](double arg_x){
419  double xx = arg_x - endpoint[i];
420  return p[i]*xx+q[i]*xx*xx/2.+r[i]*xx*xx*xx/3.+s[i]*xx*xx*xx*xx/4.;
421  };
422  res += F(sub_x1)-F(sub_x0);
423  if(endpoint[i+1] >= arg_x1) { fsum = 2; }
424  }
425  }
426  return res;
