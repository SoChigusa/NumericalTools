234 double NTools::SimpsonIntegrator::integrate(std::function<double(double)> arg_f,
235  double arg_x0, double arg_x1, double arg_dx)
236 {
237  double res = 0.;
238  double xmd;
239  double xlg = arg_x0;
240  double fsm, fmd;
241  double flg = arg_f(arg_x0);
242  while(xlg < arg_x1) {
243  xmd = xlg + 0.5 * arg_dx;
244  xlg = xlg + arg_dx;
245  fsm = flg;
246  fmd = arg_f(xmd);
247  flg = arg_f(xlg);
248  res += (fsm+4.*fmd+flg)*arg_dx/6.;
249  }
250  return res;
251 }

266 {
267  const int nmesh = arg_x.size();
268  if(arg_f.size() != nmesh) {
269  throw "Mismatch between # of mesh of x and y\n";
270  }
271 
272  double res = 0.;
273  if(arg_flag == "Simpson") {
274  if(nmesh % 2 == 0) {
275  throw "Odd # of date including start and end point is needed\n";
276  } else if(nmesh < 3) {
277  throw "Too small # of mesh\n";
278  }
279 
280  const double dx = arg_x[2] - arg_x[0];
281  double xmd = arg_x[1];
282  for(int i = 1; i < nmesh-1; i=i+2) {
283  if(fabs(arg_x[i-1]-(xmd-0.5*dx)) > 1.e-10 ||
284  fabs(arg_x[i]-xmd) > 1.e-10) {
285  throw "Non-universal choice of dx\n";
286  }
287  res += (arg_f[i-1]+4.*arg_f[i]+arg_f[i+1])*dx/6.;
288  xmd = xmd+dx;
289  }
290  }
291 
292  else if(arg_flag == "Spline") {
293  NTools::SplineInterpolator mySpline;
294  mySpline.interpolate(arg_x, arg_f);
295  res = mySpline.integrate();
296  }
297 
298  else {
299  throw "Unsupported flag for SimpsonIntegrator\n";
300  }
301  return res;

312 double NTools::BooleIntegrator::integrate(double arg_dx, std::vector<double> arg_f) {
313  double res = 0.;
314  const int nmesh = arg_f.size();
315  static const double f29o24 = 29./24.;
316  static const double f71o24 = 71./24.;
317  static const double f35o24 = 35./24.;
318  static const double to45 = 2./45.;
319  res += f29o24*arg_f[0]+f71o24*arg_f[1]+f35o24*arg_f[2]+0.375*arg_f[3]; // 1st, 2nd, 3rd order
320  res += 0.375*arg_f[nmesh-4]+f35o24*arg_f[nmesh-3]+f71o24*arg_f[nmesh-2]+f29o24*arg_f[nmesh-1];
321  for(int i = 2; i < nmesh-2; ++i) {// 1st order

323  }
324  res *= 0.25 * arg_dx; // treat over-counting & spacing
325  return res;

335 void NTools::SplineInterpolator::interpolate(std::vector<double> arg_x, std::vector<double> arg_y)
336 {
337  const int n = arg_x.size()-1;
338  if(arg_y.size() != n+1) {
339  throw "Mismatch between # of x and y points\n";
340  }
341 
342 #ifdef _NTOOLS_CHECK_ASCEND
343  else {
344  std::vector<double> check;
345  std::copy(arg_x.begin(), arg_x.end(), std::back_inserter(check));
346  std::sort(check.begin(), check.end());
347  if(!std::equal(arg_x.begin(), arg_x.end(), check.begin())) {
348  throw "Data should be ascending order in x\n";
349  }
350  }
351 #endif
352 
353  endpoint.resize(arg_x.size());
354  std::copy(arg_x.begin(), arg_x.end(), endpoint.begin());
355  std::vector<double> h(n), b(n), d(n), g(n), u(n);
356  p.resize(n);
357  q.resize(n);
358  r.resize(n+1);
359  s.resize(n);
360 
361  int i;
362  for(i=0; i<n; i++) { h[i]=arg_x[i+1]-arg_x[i]; }
363  for(i=1; i<n; i++) {
364  b[i]=2.*(h[i]+h[i-1]);
365  d[i]=3.*((arg_y[i+1]-arg_y[i])/h[i] - (arg_y[i]-arg_y[i-1])/h[i-1]);
366  }
367  g[1]=h[1]/b[1];
368  for(i=2; i<n-1; i++) { g[i]=h[i]/(b[i]-h[i-1]*g[i-1]); }
369  u[1]=d[1]/b[1];
370  for(i=2; i<n; i++) { u[i]=(d[i]-h[i-1]*u[i-1])/(b[i]-h[i-1]*g[i-1]); }
371  r[0]=0; r[n]=0;
372  r[n-1]=u[n-1];
373  for(i=n-2; i>=1; i--) { r[i]=u[i]-g[i]*r[i+1]; }
374  for(i=0; i<n; i++) {
375  p[i]=arg_y[i];
376  q[i]=(arg_y[i+1]-arg_y[i])/h[i]-h[i]*(r[i+1]+2.*r[i])/3.;
377  s[i]=(r[i+1]-r[i])/(3.*h[i]);
378  }
379 }

420 double NTools::SplineInterpolator::integrate(double arg_x0, double arg_x1)
421 {
422  if(arg_x0 < endpoint.front() || arg_x0 > endpoint.back() ||
423  arg_x1 < endpoint.front() || arg_x1 > endpoint.back()) {
424  throw "Point is not between the interpolation region in SplineInterpolator::integrate\n";
425  } else if(arg_x0 > arg_x1) {
426  throw "Integration with x0 > x1 is not supported\n";
427  }
428 
429  bool fsum = 0;
430  double res = 0.;
431  double sub_x0, sub_x1;
432  const int n = endpoint.size()-1;
433  for(int i = 0; i < n; i++) {
434  if(fsum == 0) {
435  if(endpoint[i+1] > arg_x0) { fsum = 1; }
436  }
437  if(fsum == 1) {
438  sub_x0 = std::max(endpoint[i], arg_x0);
439  sub_x1 = std::min(endpoint[i+1], arg_x1);
440  auto F = [=](double arg_x){
441  double xx = arg_x - endpoint[i];
442  return p[i]*xx+q[i]*xx*xx/2.+r[i]*xx*xx*xx/3.+s[i]*xx*xx*xx*xx/4.;
443  };
444  res += F(sub_x1)-F(sub_x0);
445  if(endpoint[i+1] >= arg_x1) { fsum = 2; }
446  }
447  }
448  return res;
